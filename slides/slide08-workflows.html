# Connect step outputs to inputs
steps:
  extract:
    run: extract-header.cwl
    in:
      fits_file: input_image  # workflow input
    out: [header_json]

  process:
    run: calculate-stats.cwl
    in:
      fits_file: input_image
    out: [stats_json]

  report:
    run: generate-report.cwl
    in:
      header: extract/header_json  # step output
      stats: process/stats_json
    out: [report]

outputs:
  final_report:
    outputSource: report/report
